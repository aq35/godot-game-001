extends Node

func _ready():
    print_tree(self)


func print_tree(node: Node, indent: int = 0) -> void:
    var line = "  " * indent + "- " + node.name + " : " + node.get_class()
    print(line)
    for child in node.get_children():
        if child is Node:
            print_tree(child, indent + 1)
